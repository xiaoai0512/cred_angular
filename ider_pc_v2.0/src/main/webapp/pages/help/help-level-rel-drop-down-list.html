级联下拉框使用 - 入门指南
级联下拉框代码与示例。
说明：第一个文本框从后台查询出数据后展现。第二个文本框监听第一个文本框中的值，在第一个文本框下拉选中后，触发请求查询，将数据展现在第二个文本框中。 同理，第三个文本框也一样。
地址下拉框。代码： <div address province="model.province" city="model.city" county="model.county" ></div> 必须声明省市元素，区域参数可缺省,每个控制域的数值获取对应选中的地区编号。
html代码
    

         
         

            
<select jf-select="dy1"  ng-model="demo.dy1">

            
<select jf-select="dy2"  ng-model="demo.dy2">

            
<select jf-select="dy3"  ng-model="demo.dy3">
js代码
    $scope.demo = {};
    $scope.dy1 ={
        type:"dynamic", //动态查询数据
        param:{pId:"10002"}, //默认查询条件
        text:"name",  //下拉框显示的文本内容
        value:"id",   //下拉框文本内容对应的值
        resource:"province.query" //数据源调用的action
    } ;
    $scope.dy2 ={
        type:"dynamic",
        watch:"demo.dy1", //该字段值改变自动触发查询做过滤
        text:"name",
        value:"id",
        resource:"city.query"
    } ;
    $scope.dy3 ={
        type:"dynamic",
        watch:"demo.dy2",
        text:"name",
        value:"id",
        resource:"area.query"
    } ;
