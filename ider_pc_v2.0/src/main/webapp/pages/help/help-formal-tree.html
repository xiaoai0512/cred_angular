js代码
    $scope.checkTreeDemo = {
        isCheck : true, //是否复选
        resource : 'roleMenu.query', //资源查询
        //单击事件
        nodeClick : function(treeNode) {
            //树节点单击对象
            console.log(treeNode);
            alert(treeNode.name);
        },
        //选中事件
        nodeCheck : function(treeNode) {
            //输出树节点选中的信息
            console.log(this.getChecked());
        }
    }

    $scope.radioTreeDemo = {
        isCheck : false,
        resource : 'roleMenu.query',
        nodeClick : function(treeNode) {
            alert(treeNode.name);
        }
    }

    //树的重新加载方法
    $scope.radioTreeDemo.reload();
